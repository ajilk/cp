# converging from both ends
left, right = 0, n - 1

while left < right:
    # process A[left] and A[right]

    if condition:
        left += 1

        right -= 1

<Tab value="Slow-Fast">
```python
# slow and fast pointers
slow = fast = start

while fast and fast.next:
    slow = slow.next
    fast = fast.next.next

    # detect cycle
    if slow == fast:
        break

# find cycle start or middle
slow = start
while slow != fast:
    slow = slow.next
    fast = fast.next
```
